from turtle import fd, pencolor, seth, penup, pendown
def Draw(length):
    pencolor("blue")
    seth(90)
    fd(length)
    seth(-30)
    fd(3*length)
    seth(-150)
    fd(3*length)
    seth(90)
    fd(2*length)
    seth(30)
    fd(2*length)
    seth(-90)
    fd(3*length)
    seth(150)
    fd(3*length)
    seth(30)
    fd(length)
Draw(50)
