from turtle import fd, pencolor, seth
def DrawTriangle(length):
    pencolor("blue")
    seth(60)
    fd(length)
    seth(-60)
    fd(length*2)
    seth(180)
    fd(length*2)
    seth(60)
    fd(length)
    seth(0)
    fd(length)
    seth(-120)
    fd(length)
    seth(120)
    fd(length)
DrawTriangle(200)
